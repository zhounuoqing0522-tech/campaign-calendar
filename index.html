<script>

const regions = {India:'asia',UK:'europe',Kazakhstan:'asia',Georgia:'europe',Spain:'europe',Turkey:'europe',Latam:'americas',Indonesia:'asia',Portugal:'europe',MENA:'europe',MiniEU:'europe',France:'europe',Germany:'europe',Africa:'europe',SEA:'asia',CzechSlovak:'europe',Korea:'asia',Netherlands:'europe',Poland:'europe',Nordics:'europe',Brazil:'americas',CIS:'europe',Vietnam:'asia',Italy:'europe',Mandarin:'asia'};

const parts={'User Growth':'growth','Monetization':'monetize','Operations':'ops'};

let campaigns=[{id:1,name:'Q2品牌推广',dept:'MarketingOps',region:'Global',part:'User Growth',start:'2026-04-01',end:'2026-04-30'},{id:2,name:'亚洲市场活动',dept:'SBU',region:'India',part:'User Growth',start:'2026-04-05',end:'2026-04-20'},{id:3,name:'欧洲运营活动',dept:'DBU',region:'UK',part:'Operations',start:'2026-04-10',end:'2026-04-25'},{id:4,name:'拉新激励',dept:'GPT',region:'Brazil',part:'User Growth',start:'2026-04-12',end:'2026-04-30'},{id:5,name:'支付产品发布',dept:'PBUCard',region:'Global',part:'Monetization',start:'2026-04-15',end:'2026-05-05'}];

let activeRegion='all',activePart='all';

function setFilter(type,value,el){if(type==='region'){activeRegion=value;document.querySelectorAll('#regionFilters .filter-btn').forEach(b=>b.classList.remove('active'));}else if(type==='part'){activePart=value;document.querySelectorAll('#partFilters .filter-btn').forEach(b=>b.classList.remove('active'));}el.classList.add('active');renderGantt();}

function renderGantt(){const dept=document.getElementById('deptFilter').value;const startDate=new Date(document.getElementById('startDate').value);const endDate=new Date(document.getElementById('endDate').value);const daysDiff=Math.ceil((endDate-startDate)/(86400000))+1;let filtered=campaigns.filter(c=>{const cStart=new Date(c.start);const cEnd=new Date(c.end);if(activeRegion!=='all'&&(c.region==='Global'?'global':regions[c.region]||'global')!==activeRegion)return false;if(activePart!=='all'&&parts[c.part]!==activePart)return false;if(dept!=='all'&&c.dept!==dept)return false;if(cEnd<startDate||cStart>endDate)return false;return true;});const statsBar=document.getElementById('statsBar');const today=new Date();const running=filtered.filter(c=>new Date(c.start)<=today&&new Date(c.end)>=today).length;statsBar.innerHTML=`<div class="stat-chip"><div class="stat-dot" style="background:#64748b"></div>共 <strong>${filtered.length}</strong> 个活动</div><div class="stat-chip"><div class="stat-dot" style="background:#ef4444"></div>进行中 <strong>${running}</strong></div><div class="stat-chip"><div class="stat-dot" style="background:#f59e0b"></div>即将开始 <strong>${filtered.length-running}</strong></div>`;const container=document.getElementById('ganttContainer');if(filtered.length===0){container.innerHTML='<div class="empty-state">🔍 没有匹配的活动</div>';return;}let html='<div class="gantt-header"><div class="gantt-labels-col">活动</div><div class="gantt-dates-header">';for(let i=0;i<daysDiff;i++){const d=new Date(startDate);d.setDate(d.getDate()+i);const day=d.getDate();const dow=d.getDay();const isWeekend=dow===0||dow===6;html+=`<div class="date-cell${isWeekend?' weekend':''}">${day}</div>`;}html+='</div></div>';const colorMap={SBU:'#3b82f6',DBU:'#22c55e',GPT:'#f59e0b',PBUCard:'#a855f7',MarketingOps:'#ec4899'};filtered.forEach(c=>{const cStart=new Date(c.start);const cEnd=new Date(c.end);const offsetStart=Math.max(0,Math.ceil((cStart-startDate)/86400000));const offsetEnd=Math.min(daysDiff,Math.ceil((cEnd-startDate)/86400000)+1);const width=((offsetEnd-offsetStart)/daysDiff)*100;const left=(offsetStart/daysDiff)*100;const color=colorMap[c.dept]||'#64748b';html+=`<div class="gantt-row"><div class="gantt-row-label"><div class="row-name">${c.name}</div><div class="row-meta">${c.dept} · ${c.region}</div></div><div class="gantt-row-bars"><div class="bar" style="left:${left}%;width:${width}%;background:${color};" onmouseenter="showTooltip(event,'${c.name}','${c.dept}','${c.region}','${c.start}','${c.end}')" onmouseleave="hideTooltip()">${width>8?c.name:''}</div></div></div>`;});container.innerHTML=html;}

function showTooltip(e,name,dept,region,start,end){const tip=document.getElementById('tooltip');tip.innerHTML=`<div class="tooltip-title">${name}</div><div class="tooltip-row">部门：<span>${dept}</span></div><div class="tooltip-row">地区：<span>${region}</span></div><div class="tooltip-row">时间：<span>${start} ~ ${end}</span></div>`;tip.classList.add('visible');tip.style.left=(e.clientX+14)+'px';tip.style.top=(e.clientY+14)+'px';}

function hideTooltip(){document.getElementById('tooltip').classList.remove('visible');}

renderGantt();

</script>
